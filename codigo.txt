using System;
float floatt()
{
    Console.WriteLine("escreva o numero: ");
    string text = Console.ReadLine();
    float res = 0;
    float.TryParse(text, out res );
    return res;
}
int inteiro()
{
    Console.WriteLine("escreva o numero: ");
    string text = Console.ReadLine();
    int res = 0;
    int.TryParse(text, out res);
    return res;
}
string sinal()
{
    Console.WriteLine("escreva o operador:| + | -| * | /");
    string texto = Console.ReadLine();
    return texto;
}
float soma(float var, float var2)
{
    float result = var + var2;
    return result;
}
float div(float var, float var2)
{
    float divisao = var / var2;
    return divisao;
}
float mult(float var, float var2)
{
    float mul = var * var2;
    return mul;
}
float sub(float var, float var2)
{
    float subb = var - var2;
    return subb;
}
